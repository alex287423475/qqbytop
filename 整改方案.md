# QQBY 官网重构 —— 完整技术落地方案

## 现状诊断总结

通过对 `qqbytop.com` 的实际抓取，我确认了以下关键问题：

**搜索引擎中的"身份污染"极为严重。** 多个子页面的 `<title>` 标签直接显示为 "永久开源免费的PHP企业网站开发建设管理系统"——这是 PbootCMS 模板的默认标题，意味着搜索引擎把贵公司网站和一个开源 CMS 模板的推广信息混为一谈。对于一个面向 B 端客户的翻译服务机构，这等同于在名片上印着别人的品牌名。

**URL 结构不规范。** 当前使用 `?biyi/`、`?news/126.html` 这种查询参数式路由，对搜索引擎极不友好，无法建立清晰的页面层级和站内权重传递。

**内容缺乏差异化。** 首页模块排布是典型的"翻译公司模板"——服务范围、语种列表、三步流程、客户评价——这些和竞品高度同质化，没有突出 QQBY 在技术本地化、跨境合规等领域的护城河。

### 2.4 i18n 路由中间件

```typescript
// src/middleware.ts
import { NextRequest, NextResponse } from "next/server";
import { i18nConfig, type Locale } from "@/lib/i18n/config";

function getPreferredLocale(request: NextRequest): Locale {
  const acceptLang = request.headers.get("accept-language") || "";
  // 简单的语言偏好检测
  if (acceptLang.includes("ja")) return "ja";
  if (acceptLang.includes("en")) return "en";
  return i18nConfig.defaultLocale;
}

export function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // 检查路径是否已包含 locale 前缀
  const hasLocale = i18nConfig.locales.some(
    (locale) => pathname.startsWith(`/${locale}/`) || pathname === `/${locale}`
  );

  if (hasLocale) return NextResponse.next();

  // 跳过静态资源和 API 路由
  if (
    pathname.startsWith("/_next") ||
    pathname.startsWith("/api") ||
    pathname.startsWith("/images") ||
    pathname.includes(".")
  ) {
    return NextResponse.next();
  }

  // 未携带 locale → 302 重定向到匹配的语种路径
  const locale = getPreferredLocale(request);
  return NextResponse.redirect(
    new URL(`/${locale}${pathname}`, request.url)
  );
}

export const config = {
  matcher: ["/((?!_next|api|images|favicon.ico).*)"],
};
```

```tsx
// 根据语种切换字体与行高
const localeStyles: Record<Locale, string> = {
  zh: "font-sans leading-zh-relaxed",
  en: "font-sans leading-en-normal",
  ja: "font-ja leading-ja-relaxed",
};

export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode;
  params: Promise<{ locale: string }>;
}) {
  const { locale } = await params;

  if (!i18nConfig.locales.includes(locale as Locale)) {
    notFound();
  }

  const dict = await getDictionary(locale as Locale);

  return (
    <html lang={locale} className={localeStyles[locale as Locale]}>
      <body className="bg-white text-brand-900 antialiased">
        <Header locale={locale as Locale} dict={dict.nav} />
        <main>{children}</main>
        <Footer locale={locale as Locale} dict={dict} />
        <FloatingContact locale={locale as Locale} />
      </body>
    </html>
  );
}
```

## 七、动态 Sitemap（助力多语言 SEO）

```typescript
// src/app/sitemap.ts
import { MetadataRoute } from "next";
import { i18nConfig } from "@/lib/i18n/config";

const BASE_URL = "https://qqbytop.com";

// 所有需要被索引的页面路径
const staticPages = [
  "",
  "/services",
  "/services/document-translation",
  "/services/legal-compliance",
  "/services/cross-border-ecommerce",
  "/services/technical-localization",
  "/industries",
  "/industries/legal",
  "/industries/ecommerce",
  "/industries/technology",
  "/industries/manufacturing",
  "/about",
  "/pricing",
  "/quote",
  "/blog",
];

export default function sitemap(): MetadataRoute.Sitemap {
  const entries: MetadataRoute.Sitemap = [];

  for (const page of staticPages) {
    for (const locale of i18nConfig.locales) {
      entries.push({
        url: `${BASE_URL}/${locale}${page}`,
        lastModified: new Date(),
        changeFrequency: page === "" ? "weekly" : "monthly",
        priority: page === "" ? 1.0 : page.startsWith("/services") ? 0.9 : 0.7,
        alternates: {
          languages: Object.fromEntries(
            i18nConfig.locales.map((l) => [l, `${BASE_URL}/${l}${page}`])
          ),
        },
      });
    }
  }

  return entries;
}
```

## 八、导航栏 Header（含语种切换器）

```tsx
// src/components/layout/Header.tsx
"use client";

import { useState } from "react";
import Link from "next/link";
import { usePathname } from "next/navigation";
import type { Locale } from "@/lib/i18n/config";
import { i18nConfig } from "@/lib/i18n/config";

interface HeaderProps {
  locale: Locale;
  dict: {
    services: string;
    industries: string;
    about: string;
    pricing: string;
    blog: string;
    quote: string;
  };
}

export function Header({ locale, dict }: HeaderProps) {
  const [mobileOpen, setMobileOpen] = useState(false);
  const pathname = usePathname();

  const navLinks = [
    { href: `/${locale}/services`, label: dict.services },
    { href: `/${locale}/industries`, label: dict.industries },
    { href: `/${locale}/about`, label: dict.about },
    { href: `/${locale}/pricing`, label: dict.pricing },
    { href: `/${locale}/blog`, label: dict.blog },
  ];

  // 语种切换：保留当前路径，仅替换 locale 前缀
  function switchLocale(newLocale: string) {
    const segments = pathname.split("/");
    segments[1] = newLocale;
    return segments.join("/");
  }

  return (
    <header className="sticky top-0 z-40 border-b border-slate-200/80 bg-white/80 backdrop-blur-lg">
      <div className="mx-auto flex max-w-7xl items-center justify-between px-6 py-4">
        {/* Logo */}
        <Link href={`/${locale}`} className="flex items-center gap-3">
          <span className="text-xl font-bold text-brand-900">
            QQBY<span className="text-brand-600">.</span>
          </span>
          <span className="hidden text-sm text-slate-500 sm:inline">
            {locale === "zh" ? "全球博译翻译" : locale === "ja" ? "グローバル翻訳" : "Global Translation"}
          </span>
        </Link>

        {/* Desktop Nav */}
        <nav className="hidden items-center gap-8 lg:flex">
          {navLinks.map((link) => (
            <Link
              key={link.href}
              href={link.href}
              className={`text-sm font-medium transition-colors ${
                pathname.startsWith(link.href)
                  ? "text-brand-600"
                  : "text-slate-600 hover:text-brand-900"
              }`}
            >
              {link.label}
            </Link>
          ))}
        </nav>

        {/* 右侧操作区 */}
        <div className="flex items-center gap-4">
          {/* 语种切换 */}
          <div className="relative">
            <select
              value={locale}
              onChange={(e) => {
                window.location.href = switchLocale(e.target.value);
              }}
              className="appearance-none rounded-md border border-slate-200 bg-transparent px-3 py-1.5 text-sm text-slate-600 focus:border-brand-600 focus:outline-none"
            >
              {i18nConfig.locales.map((l) => (
                <option key={l} value={l}>
                  {i18nConfig.localeNames[l]}
                </option>
              ))}
            </select>
          </div>

          {/* CTA 按钮 */}
          <Link
            href={`/${locale}/quote`}
            className="hidden rounded-lg bg-brand-600 px-5 py-2.5 text-sm font-semibold text-white shadow-sm transition-all hover:bg-brand-500 sm:inline-flex"
          >
            {dict.quote}
          </Link>

          {/* Mobile hamburger */}
          <button
            onClick={() => setMobileOpen(!mobileOpen)}
            className="rounded-md p-2 text-slate-600 lg:hidden"
            aria-label="Toggle navigation"
          >
            <svg className="h-6 w-6" fill="none" viewBox="0 0 24 24" strokeWidth={1.5} stroke="currentColor">
              {mobileOpen ? (
                <path strokeLinecap="round" strokeLinejoin="round" d="M6 18 18 6M6 6l12 12" />
              ) : (
                <path strokeLinecap="round" strokeLinejoin="round" d="M3.75 6.75h16.5M3.75 12h16.5m-16.5 5.25h16.5" />
              )}
            </svg>
          </button>
        </div>
      </div>

      {/* Mobile Nav Drawer */}
      {mobileOpen && (
        <div className="border-t border-slate-200 bg-white px-6 py-4 lg:hidden">
          <nav className="flex flex-col gap-4">
            {navLinks.map((link) => (
              <Link
                key={link.href}
                href={link.href}
                onClick={() => setMobileOpen(false)}
                className="text-base font-medium text-slate-700"
              >
                {link.label}
              </Link>
            ))}
            <Link
              href={`/${locale}/quote`}
              onClick={() => setMobileOpen(false)}
              className="mt-2 rounded-lg bg-brand-600 px-5 py-3 text-center text-base font-semibold text-white"
            >
              {dict.quote}
            </Link>
          </nav>
        </div>
      )}
    </header>
  );
}
```
